ACTOR WSDoomMarine : DoomPlayer
{
  Speed 1.0
  Health 50
  Radius 16
  Height 56
  Mass 100
  PainChance 255
  Player.DisplayName "Marine"
  Player.CrouchSprite "PLYC"
  Player.StartItem "PistolWS"
  Player.StartItem "wepslot1"
  Player.StartItem "FistWS"
  Player.StartItem "Clip", 50
  Player.StartItem "Pistmag", 15
  Player.StartItem "SGmag", 8
  Player.StartItem "Playerlevel", 1
  Player.StartItem "SPHeal"
  
  Player.StartItem "StatVit", 4
  Player.StartItem "StatATK", 4
  Player.StartItem "StatINT", 4
  Player.StartItem "StatSTR", 4
  Player.StartItem "StatAGI", 4
  
  //Player.StartItem "RocketAmmo", 50
  //Player.StartItem "RocketLauncherWS"
  //Player.StartItem "PlasmaRifleWS"
  //Player.StartItem "Cell" 300
  Player.ColorRange 112, 127
  Player.ColorSet 0, "Green",         0x70, 0x7F,  0x72
  Player.ColorSet 1, "Gray",          0x60, 0x6F,  0x62 // Called "Indigo" originally so as to have a unique initial
  Player.ColorSet 2, "Brown",         0x40, 0x4F,  0x42
  Player.ColorSet 3, "Red",           0x20, 0x2F,  0x22
  // Doom Legacy additions
  Player.ColorSet 4, "Light Gray",    0x58, 0x67,  0x5A
  Player.ColorSet 5, "Light Brown",   0x38, 0x47,  0x3A
  Player.ColorSet 6, "Light Red",     0xB0, 0xBF,  0xB2
  Player.ColorSet 7, "Light Blue",    0xC0, 0xCF,  0xC2

  var int User_weapon;
  var int User_Spell;
  
  States
  {
  Spawn:
	PLAY A -1
    Loop
  See:
    PLAY ABCD 4 
    Loop
  Missile:
    PLAY E 12
    Goto Spawn
  Melee:
    PLAY F 6 BRIGHT
    Goto Missile
  Pain:
    TNT1 A 0 A_GiveInventory("StatProgressVIT",Random(1,8))
	PLAY G 4 
    PLAY G 4 A_Pain
    Goto Spawn
  Death:
    PLAY H 0 A_TakeInventory("HasLoaded")
	PLAY H 0 A_PlayerSkinCheck("AltSkinDeath")
  Death1:
    PLAY H 10
    PLAY I 10 A_PlayerScream
    PLAY J 10 A_NoBlocking
    PLAY KLM 10
    PLAY N -1
    Stop
  XDeath:
    PLAY O 0 A_TakeInventory("HasLoaded")
	PLAY O 0 A_PlayerSkinCheck("AltSkinXDeath")
  XDeath1:
    PLAY O 5
    PLAY P 5 A_XScream
    PLAY Q 5 A_NoBlocking
    PLAY RSTUV 5
    PLAY W -1
    Stop
  AltSkinDeath:
    PLAY H 6
    PLAY I 6 A_PlayerScream
    PLAY JK 6
    PLAY L 6 A_NoBlocking
    PLAY MNO 6
    PLAY P -1
    Stop
  AltSkinXDeath:
    PLAY Q 5 A_PlayerScream
    PLAY R 0 A_NoBlocking
    PLAY R 5 A_SkullPop
    PLAY STUVWX 5
    PLAY Y -1
    Stop
  }
}
