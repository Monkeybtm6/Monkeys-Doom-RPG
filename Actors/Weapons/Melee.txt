ACTOR FistWS : Fist Replaces Fist
{
  Weapon.SelectionOrder 3700
  Weapon.SlotNumber 1
  Weapon.Kickback 100
  Obituary "$OB_MPFIST"
  Tag "$TAG_FIST"
  +WEAPON.WIMPY_WEAPON
  +WEAPON.MELEEWEAPON
  States
  {
  Ready:
    PUNG A 1 A_WeaponReady
    Loop
  Deselect:
    PUNG A 1 A_Lower
    Loop
  Select:
    PUNG A 1 A_Raise
    Loop
  Fire:
    PUNG B 3
    TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, "Berserked")
  
  Normal1:
    PUNG C 4 A_CustomPunch(ACS_NamedExecuteWithResult("DAMAGESCR",10,0,0), TRUE)
    Goto FireEnd
  
  Berserked:
    PUNG C 4 A_CustomPunch(ACS_NamedExecuteWithResult("DAMAGESCR",40,0,0), TRUE)
  
  FireEnd:
    PUNG D 4
    PUNG C 3
    PUNG B 4 A_ReFire
    Goto Ready
  }
}

ACTOR ChainsawWS : Chainsaw Replaces Chainsaw
{
  Weapon.Kickback 0
  Weapon.SlotNumber 1
  Weapon.SelectionOrder 2200
  Weapon.UpSound "weapons/sawup"
  Weapon.ReadySound "weapons/sawidle"
  Inventory.PickupMessage "$GOTCHAINSAW"
  Obituary "$OB_MPCHAINSAW"
  Tag "$TAG_CHAINSAW"
  +WEAPON.MELEEWEAPON
  States
  {
  Ready:
    SAWG CD 4 A_WeaponReady
    Loop
  Deselect:
    SAWG C 1 A_Lower
    Loop
  Select:
    SAWG C 1 A_Raise
    Loop
  Fire:
    SAWG AB 4 A_Saw("weapons/sawfull","weapons/sawhit",ACS_NamedExecuteWithResult("DAMAGESCR",3,0,0))
    SAWG B 0 A_ReFire
    Goto Ready
  Spawn:
    CSAW A -1
    Stop
  }
}
