ACTOR MinigunWS : Weapon
{
	Game Doom
	Weapon.AmmoType "RifleAmmo" 
	Weapon.AmmoType2 "RifleAmmo" 
	Weapon.AmmoGive 0
	Weapon.AmmoUse 1
	Weapon.Selectionorder 700
	Weapon.Kickback 100
	Weapon.SlotNumber 3
	Tag "Minigun"
	Inventory.PickupMessage "$PICKUP_MINIGUN"
	Obituary "$OB_MINIGUN"
	AttackSound "weapons/minigun"
	SpawnID 214
	Radius 20
	Height 16
	States
	{
	Spawn:
		MNGN A -1
		Loop
	Ready:
		MNGG A 1 A_WeaponReady(WRF_ALLOWZOOM)
		Loop
	Deselect:
		MNGG A 1 A_Lower
		Loop
	Select: 
		MNGG A 1 A_Raise 
		Loop 
	Fire: 
		MNGG A 0 A_GunFlash
		MNGG A 2 A_FireBullets(5.6, 0, 1, (ACS_NamedExecuteWithResult("DAMAGESCR",12,0,0)), "BulletPuff")
		MNGG B 2 A_FireBullets(5.6, 0, 1, (ACS_NamedExecuteWithResult("DAMAGESCR",12,0,0)), "BulletPuff")
		MNGG A 2 A_ReFire
		MNGG B 2
		MNGG A 4
		MNGG B 4
		MNGG A 8
		MNGG B 8
		goto Ready 
	Flash: 
		MNGF A 3 BRIGHT A_Light1 
		MNGF B 3 BRIGHT A_Light0
		Stop 
		
    Zoom:
	    TNT1 A 0 A_SpawnItemEx("MinigunPickup",45,0,30,5)
	    TNT1 A 0 A_Takeinventory("Wepslot2",1)
	    TNT1 A 1 A_Takeinventory("MinigunWS",1)
	    stop
		
	}
}

Actor "MinigunPickup" : CustomInventory Replaces Minigun
{
  Inventory.PickupMessage "Picked up Minigun"
  Inventory.PickupSound "misc/W_pkup"
  states
  {
  Spawn:
    MNGN A -1
    Stop
  Pickup:
    TNT1 A 0 A_Jumpifinventory("MinigunWS",0,"PickupFail")
    TNT1 A 0 A_Jumpifinventory("Wepslot2",1,"PickupFail")
	TNT1 A 0 A_GiveInventory ("MinigunWS")
	TNT1 A 0 A_GiveInventory ("Wepslot2")
    stop
  PickupFail:
    TNT1 A 0 A_Print("Slot 2 full/Already have weapon.",1)
	Fail
  }
}
