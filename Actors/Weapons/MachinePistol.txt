//pistol

ACTOR MachinePistolWS : Weapon
{
  Weapon.SelectionOrder 1900
  Weapon.SlotNumber 2
  Weapon.AmmoUse 1
  Weapon.AmmoUse2 1
  Weapon.AmmoGive2 20
  Weapon.AmmoType "MPistmag"
  Weapon.AmmoType2 "Clip"
  Obituary "$OB_MPPISTOL"
  //+WEAPON.WIMPY_WEAPON
  +Weapon.Ammo_Optional
  Inventory.Pickupmessage "$PICKUP_PISTOL_DROPPED"
  Tag "$TAG_PISTOL"
  States
  {
  Ready:
    MPIS A 1 A_WeaponReady(wrf_allowreload)
    Loop
  Deselect:
    MPIS A 1 A_Lower
    Loop
  Select:
    MPIS A 1 A_Raise
    Loop
  Fire:
    MPIS A 0 A_JumpIfNoAmmo("Reload")
    MPIS B 1
    MPIS A 0 //ACS_NamedExecuteAlways("CheckmonHealth")
	MPIS A 0 A_FireBullets (5.6, 0, 1, (ACS_NamedExecuteWithResult("DAMAGESCR",10,0,0)), "PistPuff", FBF_USEAMMO|FBF_NORANDOM)
    MPIS A 0 A_PlaySound("weapons/PSfire", CHAN_WEAPON)
    MPIS A 1 A_GunFlash
    MPIS CD 1
    MPIS A 0 A_ReFire
	MPIS EA 1
    Goto Ready

   Reload:
    MPIS A 0 A_JumpIfInventory ("mpistmag", 20, 2)
    MPIS A 0 A_JumpIfInventory("Clip", 1, "ReloadStep")
    MPIS A 0
    goto ready
   Reloadstep:
    MPIR A 2
	MPIR BCD 2
    MPIR D 3
	MPIR D 1 A_playSound("weapons/pistolclipout", CHAN_WEAPON)
	MPIR D 16
   ReloadLoop:
    TNT1 A 0 A_TakeInventory("Clip", 1)
    TNT1 A 0 A_GiveInventory("mpistmag", 1)
    TNT1 A 0 A_JumpIfInventory("mpistmag", 20, "ReloadDone")
    TNT1 A 0 A_JumpIfInventory("Clip", 1, "ReloadLoop")
    Goto ReloadDone
   ReloadDone:
    MPIR D 1 A_playSound("weapons/pistolclipin", CHAN_WEAPON)
    MPIR D 3
	MPIR DCBA 2
    goto ready
  
  Flash:
    MPIS A 0
    MPIS A 0 Bright A_Light1
    Goto LightDone

  Spawn:
    MPIS F -1
    Stop
  }
}
