ACTOR RocketLauncherWS : RocketLauncher Replaces RocketLauncher
{
  Weapon.SelectionOrder 2500
  Weapon.SlotNumber 4
  Weapon.AmmoUse 1
  Weapon.AmmoUse2 1
  Weapon.AmmoGive 2
  Weapon.AmmoType "RocketAmmo"
  Weapon.AmmoType2 "RocketAmmo"
  +WEAPON.NOAUTOFIRE
  decal Scorch
  Inventory.PickupMessage "$GOTLAUNCHER"
  Tag "$TAG_ROCKETLAUNCHER"
  States
  {
  Ready:
    MISG A 1 A_WeaponReady
    Loop
  Deselect:
    MISG A 1 A_Lower
    Loop
  Select:
    MISG A 1 A_Raise
    Loop
  Fire:
    TNT1 A 0 //A_JumpIfInventory ("PlayerLevel", 5,"Fireplus1")
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketPlus1",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
  Fireplus1:
    TNT1 A 0 A_JumpIfInventory ("PlayerLevel", 7,"Fireplus2")
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketPlus1",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
  Fireplus2:
    TNT1 A 0 A_JumpIfInventory ("PlayerLevel", 9,"Fireplus3")
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketPlus2",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
  Fireplus3:
    TNT1 A 0 A_JumpIfInventory ("PlayerLevel", 11,"Fireplus4")
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketPlus3",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
  Fireplus4:
    TNT1 A 0 A_JumpIfInventory ("PlayerLevel", 13,"Fireplus5")
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketPlus4",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
  Fireplus5:
    TNT1 A 0
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketPlus5",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
  
  AltFire:
    SHTG A 0 A_JumpIfInventory ("PlayerLevel", 9,"AltFireplus1")
    SHTG A 0 A_PrintBold("Must be atleast Lvl 9")
	Goto Ready
  AltFireplus1:
    TNT1 A 0 A_JumpIfInventory ("PlayerLevel", 11,"AltFireplus2")
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketAltPlus1",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
	
  AltFireplus2:
    TNT1 A 0
	MISG B 8 A_GunFlash
    MISG B 12 A_FireCustomMissile ("RocketAltPlus2",0,1,0,0,0)
    MISG B 0 A_ReFire
    Goto Ready
	
  Flash:
    MISF A 3 Bright A_Light1
    MISF B 4 Bright
    MISF CD 4 Bright A_Light2
    Goto LightDone
  Spawn:
    LAUN A -1
    Stop
  }
}

ACTOR RClusterbomb
{
  Radius 8
  Height 8
  Speed 10
  Projectile
  -NOGRAVITY
  +RANDOMIZE
  +DEHEXPLOSION
  BounceType "Doom"
  BounceFactor 2
  BounceCount 2
  Gravity 0.50
  States
  {
  Spawn:
    ESES A -1 Bright
    Loop
  Death:
    HCBP A 4 Bright A_Explode(10, 64)
    HCBP B 3 Bright
    HCBP C 2 Bright
    Stop
  }
}
