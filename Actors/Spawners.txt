Actor DynSpawner
{
+NOTDMATCH
States
  {
  Spawn:
   TNT1 A 0
   TNT1 A 0
   TNT1 A 0 A_Jump(256,"DynSpawner")
  DynSpawner:
      TNT1 A 0 A_JumpIf(ACS_NamedExecuteWithResult("MapnumCheck") >= 16, "MidLevelSpawn")
	  TNT1 A 0 A_JumpIf(ACS_NamedExecuteWithResult("MapnumCheck") >= 8, "LowLevelSpawn")
	  TNT1 A 0 A_JumpIf(ACS_NamedExecuteWithResult("MapnumCheck") >= 1, "EarlyLevelSpawn")
   EarlyLevelSpawn:
	  TNT1 A 0 
	  TNT1 A -1
	  Stop
   LowLevelSpawn:
      TNT1 A 0 
	  TNT1 A -1
	  Stop
   MidLevelSpawn:
      TNT1 A 0 
	  TNT1 A -1
	  Stop
	  }
}
