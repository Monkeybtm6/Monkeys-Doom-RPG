actor WSDarkImp : DoomImp  // [RT] Inherits from the DoomImp for convenience
{
  obituary "$OB_DARKIMP"
  hitobituary "$OB_DARKIMP_MELEE"
  health 120
  meleesound "imp/melee"
  states
  {
  Spawn:
    TNT1 A 0 A_Giveinventory("StatATK",3)
	TNT1 A 0 A_Giveinventory("StatVIT",5)
	TNT1 A 1 ACS_NamedExecuteAlways("MonsterStatsint")
  Spawn2:
    DIMP AB 10 A_Look
    loop
  See:
    DIMP AABBCCDD 3 A_Chase
    loop
  Melee:
  Missile:
    DIMP EF 8 A_FaceTarget
    DIMP G 6 A_CustomComboAttack("DarkImpBall", 32,(ACS_NamedExecuteWithResult("MonDAMAGESCR",4,0,0) * random(1,8)), "imp/melee")
    goto See
  Pain:
    DIMP H 2
    DIMP H 2 A_Pain
    goto See
  Death:
    TNT1 A 0 A_GiveToTarget("Essencegiverimp", 1)
	TNT1 A 0 A_GiveToTarget("Essencegiverimp", 1)
	TNT1 A 0 A_GiveToTarget("StatProgressATK",Random(1,3))
	DIMP I 8
    DIMP J 8 A_Scream
    DIMP K 6
    DIMP L 6 A_NoBlocking
    DIMP M -1
    stop
  XDeath:
    TNT1 A 0 A_GiveToTarget("Essencegiverimp", 1)
	TNT1 A 0 A_GiveToTarget("Essencegiverimp", 1)
	TNT1 A 0 A_GiveToTarget("StatProgressATK",Random(1,3))
	DIMP N 5
    DIMP O 5 A_XScream
    DIMP P 5
    DIMP Q 5 A_NoBlocking
    DIMP RST 5
    DIMP U -1
    stop
  Raise:
    DIMP ML 8
    DIMP KJI 6
    goto See
  }
}
