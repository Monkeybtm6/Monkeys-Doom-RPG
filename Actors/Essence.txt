Actor PlayerLevel : Inventory
{          
  Inventory.PickupMessage "Gained Level!!!"
  Inventory.Amount 1
  Inventory.MaxAmount 60
  +INVENTORY.ALWAYSPICKUP
  +NOGRAVITY
  States
  {
  Spawn:
    ESES A -1 
    loop
	Pickup:
    TNT1 A 0 ACS_NamedExecuteAlways("UpdateStats")
	stop
  }
}
